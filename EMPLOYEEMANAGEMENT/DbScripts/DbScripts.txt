create database [EMPDB];
use [EMPDB];
-- Create Tables
CREATE TABLE STATE (
    STATEID INT IDENTITY(1,1) PRIMARY KEY,
    STATENAME NVARCHAR(MAX) NULL
);

CREATE TABLE GENDER (
    GENDERID INT IDENTITY(1,1) PRIMARY KEY,
    GENDERNAME VARCHAR(10) NULL
);

CREATE TABLE CITY (
    CITYID INT IDENTITY(1,1) PRIMARY KEY,
    CITYNAME NVARCHAR(MAX) NULL,
    STATEID INT NULL,
    FOREIGN KEY (STATEID) REFERENCES STATE(STATEID)
);

CREATE TABLE DEPARTMENT (
    DEPTID INT IDENTITY(1,1) PRIMARY KEY,
    DEPTNAME VARCHAR(100) NULL
);

CREATE TABLE EMPLOYEE (
    EMPID INT IDENTITY(1,1) PRIMARY KEY,
    EMPNAME NVARCHAR(MAX) NULL,
    EMPEMAIL NVARCHAR(MAX) NULL,
    EMPPHONE CHAR(10) NULL,
    EMPADDRESS NVARCHAR(MAX) NULL,
    EMPGENDER INT NULL,
    EMPSTATE INT NULL,
    EMPCITY INT NULL,
    EMPDEPT INT NULL,
    ACTIVEFLAG CHAR(1) NULL,
    DELETEFLAG CHAR(1) NULL,
    EMPSAL MONEY NULL,
    FOREIGN KEY (EMPGENDER) REFERENCES GENDER(GENDERID),
    FOREIGN KEY (EMPSTATE) REFERENCES STATE(STATEID),
    FOREIGN KEY (EMPCITY) REFERENCES CITY(CITYID),
    FOREIGN KEY (EMPDEPT) REFERENCES DEPARTMENT(DEPTID)
);
